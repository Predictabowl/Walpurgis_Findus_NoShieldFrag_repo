version "3.0"

class WalpShieldPhysicsFix : EventHandler
{
    override void WorldThingSpawned(WorldEvent e)
    {
        if (!e.Thing) return;

        Name cls = e.Thing.GetClassName();

        // Option 1: Fix the Monsters by dynamically injecting the Species match
        if (cls == 'CentaurSergeantW' || cls == 'CentaurEliteW' || 
            cls == 'OrkYoof' || cls == 'OrkNob' || cls == 'CentaurMashItWalp')
        {
            e.Thing.Species = "Centaur";
            e.Thing.bThruspecies = true;
        }

        // Option 2 (Double Safety): Dynamically strip the SOLID flag from the shields so they can never push-telefrag
        if (cls == 'CentaurShieldHitbox' || cls == 'CentaurLeaderShieldHitbox' || 
            cls == 'OrkShieldHitbox' || cls == 'OrkLeaderShieldHitbox')
        {
            e.Thing.bSolid = false;
        }
    }
}
